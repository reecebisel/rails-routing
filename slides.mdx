### CREATE

```ruby
class PostsController < ApplicationController
  ...

  def new
    @post = Post.new
  end

  def create
    @post = Post.new(post_params)

    if @post.save
      redirect_to :index, notice: "NOICE"
    else
        flash.now[:error] @post.errors.full_messages
      render :new
    end
  end
end
```
